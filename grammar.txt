
<line> ::= <stmt> "."
<stmt> ::= <expr> |
           <name> <name> <expr> { "," <expr> } |
           <name> ":=" <stmt> |
           <name> <name> ":=" <stmt>
<expr> ::= "(" <stmt> ")" | <name> | <name> <name>

========

<line> ::= <stmt> "."
<stmt> ::= <qname> <stmt-rest> |
<stmt-rest> ::= ":=" <stmt> |
                <name> <stmt-rhs> |
                <lambda>
<stmt-rhs> ::= ":=" <stmt> |
               <expr> { "," <expr> } |
               <lambda>
<expr> ::= <qname> <expr-rhs>
<expr-rhs> ::= <name> |
               <lambda>
<qname> ::= <name> |
            "(" <stmt> ")"

========

<line> ::= <stmt> "."
<stmt> ::= <chain> <name> <rhs> |
           <literalish>
<rhs> ::= <lambda> |
          <literalish> |
          ":=" <stmt> |
          ":" <arglist> |
          "=" <stmt> |
          <literalish> "=" <stmt>
<arglist> ::= <lambda> |
              <arg> <arglist1>
<arglist1> ::= <lambda> |
               "," <arg> <arglist1>
<arg> ::= <nonemptychain>
<nonemptychain> ::= <chain> <name> |
                    <chain> <name> <literalish> |
                    <literalish>
<chain> ::= <nonemptychain> |
            <lambda>
<literalish> ::= "~" <text> <literalish> |
                 "(" <stmt> ")" |
                 <literal>
<literal> ::= "{" <linelist> "}" |
              <number> |
              "\"" <text> "\"" |
              "'" <text> |
              "'" "(" <text> ")" |
              "~" <text> |
              "[" <arglist> "]" |
              "'" "[" <literallist> "]" |
              "#" "\"" <text> "\"" "#" |
              "#" "(" <text> ")"
<linelist> ::= <line> <linelist> |
               <lambda>
<literallist> ::= <listlit> <literallist1> |
                  <lambda>
<literallist1> ::= <lambda> |
                   "," <listlit> <literallist1>
<listlit> ::= "\"" <text> "\"" |
              "'" <text> |
              "~" <text> |
              <number> |
              "'" "(" <text> ")" |
              "[" <literallist> "]" |
              <name>
